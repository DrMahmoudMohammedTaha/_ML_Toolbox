

# build app into image
docker build -t fastapi-app .

# run docker image
docker run -it ubuntu

# run and bind 8000 port on image to host 8000 port
docker run -d --name fastapi-container -p 8000:8000 fastapi-app

# run docker image in interactive mode 
docker run -it yourimage /bin/bash

# to remove all garbage and unused images
docker system prune

# to publish your image
docker login
docker tag imageName userName/imageName
docker push userName/imageName

# one command for every thing
docker init

# use compose.yaml to add important services for docker
docker compose up

# to add files and folders to be ignoredd
.dockerignore
